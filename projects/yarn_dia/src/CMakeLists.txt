set(yarn_dia_files "")
file(GLOB_RECURSE yarn_dia_files RELATIVE
    "${CMAKE_CURRENT_SOURCE_DIR}/"
    "${CMAKE_CURRENT_SOURCE_DIR}/*.cpp")

add_library(yarn_dia STATIC ${yarn_dia_files})
target_link_libraries(yarn_dia
    dynamic
    yarn
    dia
    utility
    ${LIBXML2_LIBRARIES}
    ${Boost_LIBRARIES})

set_target_properties(yarn_dia
    PROPERTIES OUTPUT_NAME dogen_yarn_dia)
set_target_properties(yarn_dia
    PROPERTIES VERSION ${DOGEN_VERSION} SOVERSION ${DOGEN_MAJOR_VERSION})

install(TARGETS yarn_dia ARCHIVE DESTINATION lib COMPONENT libraries)
